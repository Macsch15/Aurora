server {
    # Nginx configuration

    location ~ /tricolore_directory/(app|storage|src) {
        deny all;
        return 405;
    }

    location /tricolore_directory {
        if ( $uri !~ /(static)/ ) {
            rewrite /tricolore_directory/([A-Za-z0-9/]+) /tricolore_directory/index.php?/$1;
        }
    }

    # Nginx configuration
}
